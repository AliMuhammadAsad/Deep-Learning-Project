خدایه رزق راکوې له خپله مټه
نوره هيڅ راسره نشته ګټه وټه
چې دې علم په خوږو یم خبر شوے
کله کله کوزېدے نه شم له کټه
دغه خوئ راله خپل رب نه دے راکړے
دا شمله مې مدام هسکه لرې خدایه
چې پښتون دې پېدا کړے یم په خټه
د انسان د محتاجۍ نه امان غواړم
لکه سپے، چې سړے خوري رزق له چټه
اباسین چې کله خپله لاره پرېښوه
نو دا سر ورله بیا ووه په ګټه
زه ئې ونیوم له سره چې ته څوک ئې؟
بیا ئې وویستم له دره چې ته څوک ئې؟
هره لاره چې مې ستا طرفته بیائې
اے زما د سوچ محوره! چې ته څوک ئې؟
ګورې زلفې دې په مخ خورې ورې دي
اے په زړه ذهن کافره، چې ته څوک ئې؟
پښتونخوا دې د عظمت سندارې وائې
اباسينه، سخنوره، چې ته څوک ئې؟
شمعې! وژل دې مناسب نه ګڼم
ګنې په سر بلې لمبې ستړې کړې
د کائنات پېټے په سر او سترګو
خدایه! دې خپل وجود مې ښپې ستړې کړې
شپنغوزې کړي په شومه دم انګازې
ما درپسې لارې او کوڅې ستړې کړې
تا ده خدائے په کور کښې یاد نه شولو
ناوخته کیږي او مې پرېږده کنه
ته ټول عمري دي بهانې ستړې کړې
ته خو دریاب یې څه کمے نه لرې
تا اباسینه جونې ښې ستړې کړې
نه شم زه منع کېدے له محبته
په ازغو باندې ګرېوان راټولوومه
راته پېښه شوه د ګل د نزدېکته
محبت مې مجبوري ده زه انسان یم
سر مې راوړو په تلې کښې دا دے راغلم
خبردار یمه د حسن له قیمته
د مستۍ نه مزد کرمه په پټو کښې
ستا د زلفو خوشبویانې ترېنه خیژي
دا جامې مې ځکه نه وینځم له بته
زړه مې روغ نه دے دا سترګې به دې پوه کړي
لېونو سره دې ناچوي انګته
دومره بار د ښائستونو څنګه یوسي
نرۍ ملا دې جول کوي له نزاکته
ته که سل قسمه خورې زما د نوم نه
ټول رانجه دې بهېدلي دي به ښکته
چې د غرونو لېونے شې اباسینه
ځان دې وویستو له کلی له محلته
د درد سره تړلے محبت دے ځکه خوږ دے
د ده هم د ښائست هومره قیمت دے ځکه خوږ دے
ګلونه او خوشبو راباندې ګران د چا په مخ دي
د ساز مې د جانان سره نسبت دے ځکه خوږ دے
اسمان چې کله وبرېښي خندا دې راپه یاد شي
باران خو ستا د حسن تلاوت دے ځکه خوږ دے
مرګے خو د جانان غېږې له تله دي خوشحالي ده
ژوندون خو له جانانه عبارت دے ځکه خوږ دے
جادو نه دے د روح د خائستونو آئینه ده
کلام کښې مې اثر د معرفت دے ځکه خوږ دے
په ټوله پښتونخوا کښې ډیر خوږ ژبي بلبلان دي
دا ستا په اباسین کښې صداقت دے ځکه خوږ دے
هر څوم زما غوندې ساده نه وي چې زیار وباسي
خلق پُښتو مُښتو په تندر ولي کار وباسي
دا استحصال دے د پرَدو که بې حسي ده زمونږ
څه چل خو شته دے چې توبې مو هر سالار وباسي
دو‍‍ئ هر منزل ته په لنډو د رسېدو قائل دي
پښتانه ځکه په سمسور پټي کښې لار وباسي
چې ترې مالیان پخپله بوټي مېوه دار وباسي
دا چې په ځائے د محبت په زر او زور پسې ځي
چې به دُرخو د پښتونخوا ورته بلۍ ساتله
آدم لګیا دے د رباب نه هغه تار وباسي
محموده چرته یې لږ راشه ننداره خو وکړه
ایاز دې ګوره تېرې کريکې په بازار وباسي
یقین مې نه شي که بل څوک دغه معیار ومومي
چې خپل وطن له کومه وینه قلمنار وباسي
یاران خو پرېږده دُشمنان ئې هم به ځان مئين کړه
ستا اباسين خلق په دي جرم له ښار وباسي
انګور سره یاري ده که د بنګ سره یاري ده
د مستو خو د ژوند د هر یو رنګ سره یاري ده
بورا په تورو شونډو د ګلونو بوسې اخلي
د شمعې چې په ډاګه د پتنګ سره یاري ده
ګلاب چې په چمن واز ګرېوان خاندي بلبل ته
د حُسن د بادشاه چې د ملنګ سره یاري ده
دا ځکه کور او ګور او پېښور درې واړه یو دي
زما د باباګانو د اورنګ سره یاري ده
جبین له غواړي ستوري او سپوږمۍ د غاړې هار له
زما د یوه داسي شوخ و شنګ سره یاري ده
ګلونه مړاوي کيږي د پښتو په ګل دره کښې
ستا څنګه اباسینه د غورځنګ سره یاري ده
ته اے د پښتو پښتنو خدایه رامدد شې
د ژوند په اباسین کښې مو نهنګ سره یاري ده
دے به قند د محنت له کوره تل خوري
اصیلان په محبت باندې پوهېږي
کمینه خلق د خپلې لارې مل خوري
یو بڅرے په میلونو لوے ځنګل خوري
سېل چنچڼې د زمري څرمن وکاږي
دا جونګړه به یو ځل دغه محل خوري
څوک چې تشه ککرۍ لکه حباب وړي
د نبۍ سیلۍ څپېړې په ګوګل خوري
سمسور باغ راله په توره خوله مغل خوري
رهبرانو راله قام داسې تاله کړو
دُشمنان به اباسين کښې لاهو کیږي
مینه وال به شیرینې ستا د غزل خوري
