اے مۀ وینه په ویښه د خوبانو، خوبونه
په غرونو به دې سر کړي لیونے، لیونیه
ړندې به دې کړي سترګې چه ډیر ډیر ورته ګورې
بې واکه وې د سترګو پلوشے لیونیه
فطرت چه پټ پټوني په پټو کښې کوي
پاګل به دې کړي داسې تماشے، لیونیه
رنګین دې جادوګر دي د دامانو ګلونه
په چل کښې به درپورې کړي لمبې لیونیه
نه کانړے څوک راولي، نه څوک ګل دې کوڅه کښې
ترڅو به په سر خاورې بادوے، لیونیه
سائله دي تصویر جوړول ګران د جانان
څه کرښې ورانوې، څې جوړوے لیونیه
بدن په زوړ قالب کښې نوي خیال ته چرته رسي
سره چې د اُوره نه شي تیر کمال ته چرته رسي
داغ په زړۀ ښکلے ښکاري، زړه په داغ د داغه ښکلي
خود حسین جانان د زنې خال ته چرته رسي
په ورک جانان پسې کور کور کوڅه کوڅه لټول
چه بنجارے نه شي سړے وصال ته چرته رسي
سپرلي، ګلشن ګل او ګلزار، حورې غلمان او جنت
د محبت د انتخاب جمال ته چرته رسي
د بوډئ ټال په رنګ کښې ډوب سړے اسمان ته وړی خو
د لونګينو د لونګو ټال ته چرته رسي
مونږه بائیللے دے نظر زمونږه لاس واړۀ دي
د محبوب د قلا لوړ دیوال ته چرته رسي
موسم که شي شمال شمال او جانانه جانانه
ولې سائله د اوربـل شـمـال تـــه چـرتـــه رســــي
زما نه خپل جانان وطن په خپل سحر کښې ورک شو
لکه چه ځان رانه په سترګو د دلبر کښې ورک شو
خال دې پرې داغ جوړ که ، بنړو دې پرې پرهر جوړ که
زړه خو مې اوس هم شته خو دا غم په پرهر کښې ورک شو
زما نظر او ستا تصویر اور د یو بل نه واخست
خدایه دا څه اوشو نظر مې په نظر کښې ورک شو
مازیګري راشي خاموش تیر شي په غرونو واوړي
شرنګ د بنګړو شړق د پائیزیبو په ګودر کښې ورک شو
ګل راله راوړۍ که ئې دے چرته پیدا کړي پکښې
رانه مات شوے یو ازغے د زړه په سر کښې ورک شو
ما د اور اور جذبو نه ډک بدن د مینې لره
مانه زما د بې وسئ په سمندر کښې ورک شو
څومره بلا بلا خوبانو پرې دعوه لرله
هغه سائل د یو منکر جانان په در کښې ورک شو
زمانه مینه واخله غاړه واخله
هر خوا چې ځې درسره دواړه واخله
ماسره هیڅ نشته بیا هم څه چې دي
جانانه دا درسره واړه واخله
پریږده بنګلې او د بنګلو ګلونه
د محبت دنیا ویجاړه واخله
ژبه د سترګو د خبرو زده که
سلام په منډه په ولاړه واخله
نن ئې په دې ټپه کښې اوې راته
په اجاره د ګودر غاړه واخله
پردي ګلونه ګلشنونه پریږده
مزه د خپل وطن د کاړه واخله
په پردو شونډو تنده نه ماتیږي
بوسه د دې شونډو د پاړه واخله
کله خوته هم لګې شونډې سپړه
سائله څه چې غواړې غواړه واخله
اولس جانان که نو د ټول اولس جانان به شې
راشه پخپله خاوره اودریږه آسمان به شې
د ژوند په شور کښې ځنې ځنې آوازونه اوره
که ناست ئې پا به څې که پاڅیدې روان به شې
په تنهایۍ کښې وې بې درہ بې دیاره سړے
د زړونو کور کلي له راشه نو ودان به شي
ستوري خالونه په تندو که د وطن د ښکلو
تیرې دې نه شي پټولے چراغــان بــه شــې
د پردي سوري نه پخپلو سرو غرمو کښې ښه ئې
د خپل پرهر د درد د سرو ګلونو ژبه زده که
ګل به پخپله شې د ګل د خاندان به شې
ماتول هیڅ نه دي مشکل د نفرتونو پولې
د مینې لار واخله یقین به شې ایمان به شې
بیا به تمبېږې د پلکونو ګزارونه درنه
د ستمګرو مینه اوکه نو سندان به شې
یې خو افغان خو د دې خپل نوم نه بُګنیږې ولې
په آئینه کښې ځان ته اوګوره افغان به شې
ته چه ازغے ئې نو ازغے خود ګل کار نه کوي
د ګلو تخم شه نو ګل او ګلستان به شې
سترګې پیدا کۀ د ژوندون د حقیقت لیدو له
ته که تیاره هم ئې د نمر غوندې روښان به شې
یا بنجاري شه یا ملنګ د پښتنو د کوڅو
سائله دا اعزاز حاصل که شاه جهان به شې
د ژوند په دار دې اودرولے یم پښتون ارمانه
په دے قاتل قاتل ماحول کښې د ژوندون ارمانه
پرون دې هم تڼئ تڼئ ګریوان وو پروت په سینه
نن دې هم سرۀ سینه په وینو د پرون ارمانه
هر څوک دې هر راټوکیدلي ګل ته اور لګوي
د پښتنو د زرغون شوي سباوون ارمانه
د دے دنیا د شب پرستو ډله لویه شوله
د هر طرف نه ستا په خونه دے شب خون ارمانه
زہ ئې په زړۀ پورې نیولے وړم اودریږمه نه
حالات زما نه جذبه تروړي د بدلون ارمانه
په زړه مې داغ اوسه مسکے اوسه چه مړاوے نه شې
لکه ریدے مې کړلې لوئې په ځیګر خون ارمانه
شپه بلا لویه ده، بلا مزل په سر پاتې دے
شوګيروم دې د خوبونو په زنګون ارمانه
سترګو مې هم دا دې په وینو سرې جامې اغوستي
زما په خپلو پاکو وینو کښې ګلګون ارمانه
غږ مې زخمي دے خو چې ورک قام به ئې واوري که نه
زما په زړه کښې د تړون او د پاڅون ارمانه
د پیر روښان د خوشحال خان د باچا خان په خاطر
ماله جذبه را که د نوي زړه چاؤدون ارمانه
یوه ټپه مې که خـو هـر غـــوږ ته مې اورسوه
زما د خپل لیلے لیلے وطن مجنون ارمانه
بیرته به راشم ستا سائل به تاته او سپارمه
ورک یم په ځان پسې د خپل ځان په لټون ارمانه
