خو چاهېڅ کولے نه شو
دَ چا ډله نه پره وه
نه رهبر نه قافله وه
نه همت نه حوصله وه
نه اُمید وو نه اسره وه
بې وسي وه مجبوري وه
دَ بُتانو بندګي وه
دردېده وو تزينده وه
لوګېده وو سوزېده وو
اخر جام دَ صبر ډک شو
یو سحر ناساپه غږ شو
یو بنده دَ خدائے پېدا شو
پېغمبر شو رهنما شو
بې اسرې بې وسيلې وو
خو سړے دَ کار سړے وو
دَ خپل خُدائے دَ شان خبر وو
لوے نبي لوے پېغمبر وو
دَ توحيد اعلان یې وکړو
دَ تجديد اعلان یې وکړو
په بتانو یې تور ووې
په غټانو یې تور ووې
دَ مظلوم هېبت یې ورک که
دَ ظالم غرور یې سپک که
سرداران یې کور او کر کړه
ټول بُتان یې کنډ کپر کړه
دَ تفريق زهر یې خام کړه
دَ توحيد شراب یې عام کړه
 وريځې راوله باران راوله
 زلفو له څنډ ورکه طوفان راوله

 لر کړه دَ مخه آئينه لرې کړه
 ما دَ حېرت دَ بیابان راوله

 سپرليه زه به دې احسان ومنم
 خو ته دَ ځان سره جانان راوله

 ځه چې واپس زړې قيصې نوې کړو
 راشه واپس هغه دوران راوله

 ماښام لمبه کړه تیارې وتښتوه
 بله ډیوه شه پتنګان راوله

 واعظه، نه په تقوی نه رغېږم
 ما له ساقي دَ روهستان راوله

 زړه نظر سترګې په جرګه درولم
 ښکلي راغونډ که رقيبان راوله
 کلے خبر دے څوک دې ونه ويني
 مه ځه سحر دے څوک دې ونه ويني

 زما دَ هر سړي نه ويره کېږي
 په چا باور دے څوک دې ونه ويني

 په هر ګوټ پېر دَ نظرونو څوکۍ
 هر خوا خطر دے څوک دې ونه ويني

 په دې شکمنه بلۍ مه ودرېګه
 وخت مازيګر دے څوک دې ونه ويني

 قېصې پرېږده لنډه خبره کوه
 لار ده، ګودر دے څوک دې ونه ويني

 خوا له مې مه راځه بدنامه به شې
 صحرا ده، غر دے څوک دې ونه ويني

 په دې کوڅه کښې مخامخ صابره
 دَ جانان ور دے څوک دې ونه ويني
 هغه مستي هغه غورزنګ نه کوي
 سترګې دې اوس دَ مينې جنګ نه کوي

 دَ پېغلتوب دَ مستۍ خېر خو دې وي
 بنګړي دې شور نه کوي شړنګ نه کوي

 زمونږ دَ ښار ښاپیرۍ اوښکې پيري
 زمونږ دَ غره غاړې لونګ نه کوي

 هغه هم ستا دَ بڼو وبوږنېږي
 چې دَ سنګزنو تورو سنګ نه کوي

 مونږ په دې لاره بلا وينې توې کړې
 خو هسې نه رڼېږي رنګ نه کوي

 دَ محبت په سيند وينځلي زړونه
 په خِیرو نه خړيږي زنګ نه کوي

 صابر دَ مينې دَ نشې اموخته
 په صابر بنګ نه لږي بنګ نه کوي
دَ ارمان ونې مې سمسورې نه کړې
 خو دَ وفا لارې مې نورې نه کړې

 دَ درد په څړيکو کښې مې وخندوه
 غم له مې نوري سترګې تورې نه کړې

 دَ انتظار دَ صحرا تنده توبه
 په ما یې هسې لمبې پورې نه کړې

 خدایه که هر سيوري ته کېښناستمه
 هیر مې دَ هغه زلفو سيورے نه کړې

 اوس مي دَ نوم دَ توره مه یرېږه
 ما درته وې چې سترګې تورې نه کړې

 دَ هر پرهر نه مې نچوړ کړه وينه
 خو دَ نظر پیالې نسکورې نه کړې

 دَ درنو سترګو درنو زلفو مينې
 هسې نه ما په زړه کمزورے نه کړې

 ځان مې په څه څه لمبو ونه سېزو
 ته مې دَ کوم کوم آسمان ستورے نه کړې

 ما پرې دَ زړه دَ وينو رنګ ونوستو
 ما دَ غزل تقاضې سپورې نه کړې
 دَ یقين کور بيخي میرات نه شو
 زړه که مې وچاودو خو مات نه شو

 مينې ته یو وخت ملامته نه شوم
 غم دې په یو موسم کښې قات نه شو

 نن دې بڼو راسره جنګ ونه کړو
 نن دې کاته راته پسات نه شو

 دَ زمانې زیاتي ته ګوره کنه
 په مونږ یې سود یو ملاقات نه شو

 په دې ښار داسې نفرتونه آباد
 دَ مينې نوغے پکښې زیات نه شو

 څه که حالاتو پرې لښکرې وکړې
 زړه مې خيبر وو سومنات نه شو

 صابره نه به وي سپرلے به نه وي
 که لېونتوب مې پکښې زیات نه شو
سنګار دَ زړه دَ پرهرونو کوو
 نن سبا مينه دَ ګُلونو کوو

 په تودو سړيکو یې وینځلي ساتو
 څومره پالنه دَ زخمونو کوو

 کر دَ ګُلونو ارمانونو کوو
 سوال دَ سپرلي دَ بارانونو کوو

 بویه چې دا ازغي ګُلونه سپړي
 بویه چې لوبې په رنګونو کوو

 چې دَ خوبونو تعبيرونه غواړو
 بیا خو به لوبې دَ سرونو کوو

 څومره بې خوږو، بې خوږو خلقو ته
 قېصې دَ زړونو دَ دردونو کوو

 یاران دَ تېرې زمانې یادوو
 ارمان دَ ورانو محلونو کوو

 که د هر ښکلي قدمونه شمارو
 خو وظيفه دَ یار دَ پلونو کوو

 هر وخت به څه نوي کوو خبره
 یوه خبره دَ کلونو کوو

 په خورو زلفو کښې مې وزنګوه
 بیا به خبرې دَ دارونو کوو

 دَ مينې سل مجبورۍ سل غمونه
 انتظار هم په انګارونو کوو
 کله ګُلاب کله لونګ راوړم
 ستا دَ کوڅې څه نا څه رنګ راوړم

 بلې نشې مې سودا ورکه نه کړه
 زم دَ ګودر دَ غاړې بنګ راوړم

 شوګيرې سترګې خوبولے نظر
 ستا دَ سنګُلې ستا دَ سنګ راوړم

 دَ محبت نوې دُنیا لټووم
 دَ زندګۍ نوے آهنګ راوړم

 ستا دَ بڼو په ژبه ځان پوهووم
 پرېږده چې زړه په وينو رنګ راوړم


 ستا خوږو سترګو ته خواږه ګورمه
 زه لېونے یم ځان له جنګ راوړم

 هغه صابر یم که په هر ځائے کښې یم
 تاله راځمه ستا قلنګ راوړم
 نه په دُعا نه په هنر کيږي
 عشق دَ سر لوبه ده په سر کېږي

 زړه مې خورېږي سترګې مه خوروه
 دَ خورو زلفو مې خطر کېږي

 ستا دَ کوڅې خاورې په سر بادووم
 په ما څه چل هر مازيګر کېږي

 زه د هر خکلي په وعده تېر وځم
 زما نه دا غلطي اکثر کېږي

 زړه که مې سل ځله دَ کاڼي کړمه
 خو په د ستا دَ مخ اثر کېږي

 اوس هم زما او ستا دَ مينې ذکر
 کله نا کله په ګودر کېږي

 دَ ښائستونو دَ سپرلي په تمه
 دَ محبت دَ فصل کر کېږي

 شپه به دَ مینې په قيصه تېروو
 نورې خبرې به سحر کېږي

 زما صابره ستا دَ مينې په نوم
 ما سره هره ورځې شر کېږي
دَ یار دَ کلي دَ کوڅې مازيګر
 ما لېونے کړي لېونے مازيګر

 ما په جنت هم بدل کړے نه دے
 دَ محبت دَ مېکدې مازيګر

 دَ زړه د درده ناقلاره وومه
 دَ اُور ترخې اوبه مې څښې مازيګر

 سم دَ ماښامه تبې ونيومه
 تا راته کړې وې خېرې مازیګر

 رقيبان وبوږنېده ویرېده
 مونږ چې شو یو بل ته نزدې مازيګر

 ستا دَ ښائست نه تاوېده لوګېده
 غزل غزل سپرلے سپرلے مازيګر

 ستا په کوڅه کې راترغاړې ووځي
 انګار انګار، لمبې لمبې مازیګر

 ځان یې ستایه دَ ځان صفت یې کوو
 زما غزل یې ما ته وې مازيګر

 زمونږ کوڅې ته دې کاته روان وې
 زما صابره ما ليدې مازيګر
د زړه د درده لېونے شوې کنه
 زما صابره ته سړے شوې کنه

 تا دَ اُلفت معجزه نه منله
 زه چې لمبه شوم ته لوګے شوې کنه

 دَ محبت ګرمي دې وليدله
 په یخو واورو کښې خولې شوې کنه

 ما چې دَ زړه دَ رڼا رنګ ورکو
 دَ انتظار لارې رڼې شوې کنه

 ما درته وې چې سترګې مه توروه
 دَ خلقو سترګې درپسې شوې کنه

 نوے الزام راباندې ولګوه
 زړې خبرې خو زړې شوې کنه

 دَ بڼو سيوري ته مې کېنولې
 صابره سترګې دې مړې شوې کنه
مړه دې ونه کاته اوده دې نه کړمه
 نسکور پرې نه وتم نشه دې نه کړم

 هسې دې وسوم لوغړن دې کړمه
 ګوګُل مې یخ نه شو لمبه دې نه کړم

 په نننۍ خندا دې ويرېدم
 زړه مې غوڅ غوڅ نه شو اره دې نه کړم

 دَ تودې غېږې دَ خوبونو یاده
 خوب دې رانه وستو دمه دې نه کړم

 بلا دې واخلم تر بلۍ رانغلې
 ماريګر تېر شو غرغره دې نه کړم

 مخې له راغلې ژړغونے دې کړم
 په مخه لاړې دلاسه دې نه کړم

 څه راته اُوایه په څه خفه یې
 څه شولې پوهه په قيصه دې نه کړم

 مينې په هر مېدان دې وګټلم
 یو ځائے دې پرې نښوم پيکه دې نه کړم

 اوس به مې بیا درته ښېرې کولې
 صابره ښه ده چې غصه دې نه کړم
که ستا په مخ ستا په کاکل نه لیکم
 بیا خو غزل نه دے غزل نه لیکم

 ما دې زما دَ زړه ازار اُوهي
 که په قامت او په اُربل نه لیکم

 ستا یې منله ستا یې ومنله
 بس دے دَ خپل زړه سره خپل نه لیکم

 بیا به دَ زړه خبرې زړه کښې ساتو
 بیا به روداد ستا دَ محفل نه لیکم

 نور پرې دَ زړه د وینو رنګ نولمه
 نوره دَ ګُل قیصه په ګُل نه لیکم

 هغه خوبونه مې په سترګو کښې دي
 څنګ به دَ خکلي په څنګُل نه لیکم
ستوري یو، ډيوې یو، مشعلونه یو
 مونږه به دَ غرونو یو خو غرونه یو

 څه حاجت زمونږه په چا څه پرواه
 خپله لېوني خپله دشتونه یو

 عشقه چې هر څه یو ستا دَ خاورې یو
 کاڼي یو غمي یو که لعلونه یو

 تل د طوفانونو سره نښتي یو
 څومره ناقلاره ارمانونه یو

 ودردېږو خوږ شو چې څوک وژاړي
 څه د لېونو خوږمنو زړونه یو

 وټوکېږو ورژېږو خاورې شو
 مونږ دَ پښتنې مېرې ګُلونه یو

 هغسې هم سوزو که رڼا کوو
 هغسې بل شوي ارمانونه یو
نوے سبق نوے کتاب غواړي
 غرسنۍ سترګې انقلاب غواړي

 نننے ژوند نننے عقل فکر
 نوے نظر نوے نصاب غواړي

 ستا دَ ښائست په نوم بدنامه مينه
 ستا دَ قامت دَ صلیب تاب غواړي

 خیال په خورو زلفو کښې وغزېدو
 ځواني مستي غواړي شراب غواړي

 سترګې څه بل رنګې ښائست لټوي
 مينه څه بل شان انتخاب غواړي

 د بې وسۍ خاموشه غلې جذبې
 لړزېده غواړي اضطراب غواړي

 یو سړي ځان ته ملامت ونه وې
 هر څوک دَ نورو احتساب غواړي
زړه به دَ مینې سمندر ګرځوو
 ستا د غم لړې به په سر ګرځوو

 په سړیکو هغسې دردونه پېرو
 زخمونه هغسې پرهر ګرځوو

 اوس هم دَ مینې لیوني خیالونه
 ستا دَ یادونو په ګودر ګرځوو

 یو ښکلے هغسې نیازبین دے زمونږ
 یو سړے هغسې په سر ګرځوو

 یو ښکلے هر وخت په نظر کښې ساتو
 یو ګُل په کوتو کښې اکثر ګرځوو

 د نفرتونو د تیارو د ویرې
 په زړه کښې پټ د مینې لمر ګرځوو

 زمونږ د شپې شوګيرې هسې نه دي
 په ښار کښې زېرے د سحر ګرځوو
دردونه په سینو کښې پرهرونه په سينو کې
 ساتو د مينې لوے لوے ارمانونه په سینو کې

 په تمه د سپرلي د رنګینو د بارانونو
 ساتلي مو تنکي تنکي زخمونه په سينو کې

 یرېږو د یارانو تنګ نظرو د خبرو
 وراسته مو کړل د ښکلو تصويرونه په سينو کې

 خوږو خوږو سندرو ته لېواله لېوني یو
 ټوکېږي مو خواږه خواږه دردونه په سينو کې

 زمونږ مخې ته څه د نفرتونو دېوالونه
 بهېږي مو د مينې دریابونه په سينو کې

 تر هغې به رواج د خاموشۍ ته خطره وي
 تر کومې چې درزېږي زمونږ زړونه په سينو کښې

 دنیا پکښې د حورو د غلمانو د شرابو
 زمونږ لېونو زړونه جنتونه په سينو کې

 د مينې د ګودر چپو وهلي یو صابره
 زمونږه هسې نه خېژي برېخونه په سينو کې
اوس یې د زړونو سوزېده زده کړه
 سترګو دې اُور لمبه کېده زده کړه

 د ګړنګونو خو به اوس پرېوځو
 غرڅنۍ تا د هوسۍ تله زده کړه

 ستا دېوالونه د نظره مه شه
 ما آسمانونو ته خاته زده کړه

 یارانو نن مې لېونے وګڼۍ
 ښکلي اشنا ښکلي کاته زده کړه

 دا مې په یو کتاب کښې نه شو لوستے
 د محبته چې مې څه زده کړه

 ساقي په داسې غم دې واړولو
 نه مې توبه زده نه مې څخه زده کړه

 زه د احساس د سمندر لامبوزن
 ما په سور اُر کښې اوسېده زړه کړه
 ستا په خورو زلفو کښې شپه کوومه
 د عمر ستړے یم دمه کوومه

 ستا د توې غېږې سودا اخیستے
 یخې سیلۍ ته شوګيره کوومه

 د پرهر ژبه مې تڼاکې شوله
 ستا د تودو شونډو قیصه کوومه

 ستا د جبین آئینه تته مه شه
 زه پکښې خپله تماشه کوومه

 ستا د بڼو د تصوور په سيوري
 لږ غوندې خوب هره غرمه کوومه

 د الزامونو د سورو لاس آزاد
 د جانان ذکر په ښکاره کوومه

 په دردېدلو تړقېدلو څړيکو
 ستا د یادونو وظيفه کوومه

 مینه منم خو پښتو نه ماتووم
 ستا نه بغېر به ګزاره کوومه
د یارانې لوظونه مه ماتوه
 که کوه مه ګُلونه مه ماتوه

 ګُله بې ځایه سترګې مه توروه
 هسې زمونږه زړونه مه ماتوه

 د احتساب وختونه تلي ساقي
 یرېږه مه جامونه مه ماتوه

 زمانې مه چېړه د زړونو سره
 د محبت ملکونه مه ماتوه

 قاصده هېرې قيصې مه یادوه
 د زړه ورغلي ورونه مه ماتوه

 د شیرینۍ شرطونه وپيژنه
 فرهاده هسې غرونه مه ماتوه

 ستا زنځیرونه ماتېدونکي نه دي
 صابره خپل لاسونه مه ماتوه
خون به مې واخلي قاتلانې به شي
 زلفې دې مه سموه ورانې به شي

 په آئینه کښې سترګې مه توروه
 ځان ته به وګوري حېرانې به شي

 زمونږ وعدې د ښکلو څڼې نه دي
 چې رنګ به واړوي ارزانې به شي

 د محبت خواري دي هیره مه شه
 لارې که ګرانې دې آسانې به شي

 ستا د ښائست تخم به نه ورکوي
 د زړونو زمکې دي ودانې به شي

 بلا دې واخلم زلفې مه خوروه
 بلا به واخلي بلاګانې به شي

 د تورو سترګو دروغژنې قيصې
 صابره مه سپینوه ګرانې به شي
 زما وجود شوې زما ساه نه شوې
 زما اشنا زما اشنا نه شوې

 ما ته خپل ځان او ځان مې ته ګڼلې
 ته د خپل ځان وې ته د چا نه شوې

 ښه ده چې دومره خو دې وپېژندم
 مخې له راغلې وارخطا نه شوې

 زه وسوسو په مخه واخستمه
 تا شونډې پرانستې ګویا نه شوې

 زه دې اُلفت ته ملامته کړمه
 د یار بلا زما بلا نه شوې

 سترګې یې بیا هم پښتنې وې کنه
 بې وفا وې خو بې حیا نه شوې

 مینې منمه چې د هر ازمېښته
 زه روستو نه شوم ته په شا نه شوې

 زړه د بلا ښيرو مې ووينځلې
 خو د وهمونو نه صفا نه شوې

 تا د صابر لیونتوب څه وليدو
 ته د صابر د زړه سودا نه شوې
حسن چې ګُلونه شي رنګونه شي
 زړونه بهارونه بهارونه شي

 څو چې هغه سترګې هغه سترګې وي
 څنګ به زمونږ زړونه زمونږ زړونه شي

 دا ښکلے د کوم سپرلي د کوم کلي
 هسې چې لږ وخاندي ګُلونه شي

 دا زخمونه دومره بې نوا نه دي
 دلته د دردونو محفلونه شي

 چا ته د دُعا په وخت کښې یاد نه وي
 ورک دې د مئینو نصیبونه شي

 ته د انتظار شپه یوه شپه ګڼې
 ما ته خو کلونه شي عمرونه شي

 ستا یادونه دومره بلا دومره ډېر
 څه مې شوګیرې څه مې خوبونه شي

 مینې مینې زړونه وینې وینې شو
 درنګ له به د وینو بارانونه شي

 مه منه د ښکلو تسلي مینې
 ورک دې د منکرو قسمونه شي

 یه صابره دې سترګو ته اوګوره
 هېر به دې جامونه جنتونه شي
څوک چې د مینې د منظره اوځي
 هغه د ژوند د سمندره اوځي

 هغه سړے به د نظره کړي څوک
 هغه د هر چا د نظړه اوځي

 ستا د یادونو د دردونو څړیکې
 پورې راپورې د ځیګړه اوځي

 خدایه په هغه زړه به څه تېرېږي
 چې بېنوا د یار د دره اوځي

 هغه سور اُور یې هغه سړیکه نه یي
 چې د مئين زړه د پرهره اوځي

 زما خودسره باغي شوے نظر
 د یار د سترګو د اثره اوځي

 چې د خورو زلفو تیارې رڼوې
 صابره اُور به دې د سره اوځي
شرابي سترګې ګُلابي رنګونه
 ما لېونے ګړي لېوني رنګونه

 ستا دَ جبین رڼا ته وپسخېدل
 د ښائستونو د سپرلي رنګونه

 په جنتونو مئين څه پېژني
 ستا د کوڅې مازيګري رنګونه

 د جانان سترګې د نظره مه شه
 په هر پرهر مې دوړوي رنګونه

 خدایه لمن یې ترېنه داغ وانخلي
 په رڼو سترګو کښې راوړي رنګونه

 زما زخمونه اماني دردونه
 زما یادونه تمامي رنګونه

 د خورو زلفو سيورے مه خوروه
 له خوشبویو نه شي خالي رنګونه

 زه ترېنه چرې بې نشې نه شومه
 زما ساقي ما له ساتي رنګونه

 سترګې مې ووځه خو ونه وينمه
 د خپل جانان په مخ پردي رنګونه
 د نازولي زړه مې ومنله
 ګنې نو ما خو مینه نه منله

 عقله اوس پرېږده چې ټکرې خورمه
 ما د مئين لېوني زړه منله

 ستا شرابي سترګې چې اُور لمبه وې
 زما توبه به خلقو څه منله

 زما خبرې ته چا غوږ نه نيوو
 هر سړي ستا خبره ومنله

 که مېخانې ته دې راوستے وے څوک
 سپرليه ما به دې هله منله

 یارانو تاسو ته ګناه نشته دے
 ما به پخپله د هغه منله

 ما درته وې چې دروغژن وي ښکلي
 صابره تا به زما نه منله
ټولې دُنیا ته مې د یو سړي صفت کړے دے
 ما په یو څو ورځو کښې څومره محبت کړے دے

 زه د دې ښار د نفرتونو د بتانو دشمن
 ما په دې ښار کښې د مئینو امامت کړے دے

 د دې کوڅو نه صبرېده زما د وسه نه ده
 ګنې یارانو راته څومره نصیحت کړے دے


 ما د هر ګوټ سره د یار د سترګو ذکر کړے
 ما د شرابو په نشه کې عبادت کړے دے

 هغه چې زه یې د ښائست په صفت نه مړېږم
 هغه سړي زما په هر ځائے کې غېبت کړے دے

 زما رقیب د محبت په معنی څه پوهېږي
 هغه خو ژوند هم په یو لوے منافقت کړے دے

 ستا د یادونو د سپرلي رنګونه نه هېروي
 ګنې نو ما خپلو زخمونو ته منت کړے دے

 د ملازو د ملامتو یې ساتلے یمه
 زما احساس زما په هر ځائے کښې عزت کړے دے

 د ډېرو ښکلو د ښائست ثنا دې وې صابره
 خو په غزل کښې دې د یو سړي صفت کړے دے
ما د ګُلونو د رنګونو دلداري کړې ده
 ما د سپرلي د انتظار سره یاري کړې ده

 ما د خپل زړه په رڼا ستوري د آسمان رڼول
 زه که زره ووم ما د لمر سره یاري کړې ده

 زما د زړه څړیکې به دومره بې اثره نه وي
 ما د زخمونو کرونده په نري سوي کړې ده

 د جانان ذکر مې په ژبه د غزل کړے دے
 د خپل ارمان قیصه مې تل په اسوېلي کړې ده

 مونږ د دُنیا د بادشاهانو بادشاهۍ ړنګ کړې
 خو یو سړي زمونږ په زړونو بادشاهي کړې ده

 ما ته د دې خلقو د زړونو د تیارو پته ده
 ما د دې کلي په کوڅو کښې ملنګي کړې ده

 زه یې د خپل وجود د خوږ په خوږو پوهه کړمه
 په ما د یار سترګو بلا مهربانې کړې ده

 ته څه خبر یې د زړه سره تا ته څه پته ده
 چې ستا صابر څومره په ګرانه شاعري کړې ده
د حسن په نشه کښې لېونے مخې له راغے
 په چا پسې چې تلم هغه سړے مخې له راغے

 زه نه یم بختور خو دا د مینې معجزه ده
 اشنا په پردي کلي کښې د شپې مخې له راغے

 نور هیڅ نشته پوهېږمه قسم مې ماتوي
 اشنا چې مې د زړه د دروازې مخې له راغے

 د هغه غزل زادو سترګو رنګ مې ورله ورکړو
 که ګُل مخې له راغے که سپرلے مخې له راغے

 ستا حال یې راته ووې ستا قیصه یې راته وکړه
 چې څوک ستا د پلو ستا د کوڅې مخې له راغے

 لګیا وو د خپل حسن په صفت نه مړېدو
 زما غزل یې غلے غلے وې مخې له راغے

 دا مینه ده جانانه که د مینې مجبوري ده
 زما ده زړه قاتل ما له ژوندے مخې له راغے

 په ډیره موده پس مې سترګې لږې شان خوږې شوې
 په دېره موده پس هغه سړے مخې له راغے

 څادر د محبت یې په سر سیورے وو روان وو
 صابر د ژوند په لاره د غرمې مخې له راغے
بې کوره شي چې څوک په بې درکو پسې ګرځي
 درانه خلق سپکېږي چې په سپکو پسې ګرځي

 زما د قام د کومې بدبختۍ به درته وایمه
 چې څوک رازملي کیږي په ټوپکو پسې ګرځي

 د زرکو د کونترو ښکار د نورو مشغله شوه
 زما د غرونو باز په توت خوروکو پسې ګرځي

 بیا څوک د پښتنو د اتفاق سندرې وایي
 دیوان یې د خوشحال په لاس خټکو پسې ګرځي

 صلاح کوي جرګه کوي غونډېږي تندرونه
 څوک جالې جوړوي ځان له په ډکو پسې ګړځي

 د هغه غرسنۍ غړۍ مې هغسې په زړه ده
 له هغسې مې فکر په کُنجکو پسې ګرځي

 د ښکلو په سینو کښې بې غرضه زړونه ګوري
 صابر بلا وهلے دے په ورکو پسې کرځي
په چمن کښې که ګُلونه که ازغي مري
 خو زما د کور ښائست زما سپرلي مري

 دا مې لویه بدبختي نه ده نو څه ده
 په پردي جنګ کښې زما د بنډ زمري مري

 د دنیا روزګار په صلح په صلاح شي
 کوم قامونه چې د جنګ په لوري ځي مري

 د دُنیا اولسه هله کړه رادو کړۍ
 سړي یو بل سره نغښتي دي سړي مري

 مونږ نه هغو خلقو بلې لمبې تاو کړې
 چې پخپله په پوټي غوندې لوګي مري

 د خوشحال په ادیره به ماتم نه وي
 سره بڅري د اېمل د انغري مري

 د اسلام د سپېځلتیا مُبلِغینو
 په کابل کښې درنه شته مسلماني مري
زما د زړه د خوږېدو رازونه مه لټوه
 اُور به درپورې شي اشنا اُرونه مه لټوه

 د خپلو سترګو د ظلمونو به دې کرکه وشي
 مه کوه مه زمه ده زړه زخمونه مه لټوه

 زما بې سره شوګېري درپورې ونه نخښلې
 زما په سترګو کښې زما خوبونه مه لټوه

 ارمانه مه کوه د خکلو زړونه ډیر خیرن دي
 ښکلو ته ګوره خو د ښکلو زړونه مه لټوه

 په دې سرونو کښې به سر وخوري د عقل رڼا
 دا د وهمونو د تیارو غارونه مه لټوه

 مینې رښتیا مه وایه مه کوه ستومانه به شې
 ځان له زما غوندې په زور غمونه مه لټوه

 په دې کوڅو کښې د اُلفت بادونه نه الوزي
 صابره دلته د اُلفت رنګونه مه لټوه
 د محبت رازونه پټ ساتمه
 په زړه د زړه دردونه پټ ساتمه

 نور د یارانو زړونه نه یخووم
 نور به د زړه زخمونه پټ ساتمه

 په دې بې لوظه موسم څه اعتبار
 ستا د وعدو ګُلونه پټ ساتمه

 د پرهرونو خولې مې وتړلې
 نور د جانان یادونه پټ ساتمه

 د دیوالونو نه مې ویره کيږي
 ستا د راتلو وختونه پټ ساتمه

 دلته د غم ژبه څوک نه پېژني
 دلتکې خپل غمونه پټ ساتمه

 د څکو خو څه وايې چې څه کفر دے
 زه دلته تش جامونه پټ ساتمه

 د ملامتو د فتوو د ویرې
 په سر کښې ډير سوالونه پټ ساتمه

 د خلقو خولې خو تا ته ښه پته ده
 صابره ستا خطونه پټ ساتمه
 د محبت د لېونو په ژبه نه پوهېږي
 ستا د ښار خلق د سړو په ژبه نه پوهېږي

 چا چې د کلي مازیګر سندره کړے نه وي
 هغه د بام د اشارو په ژبه نه پوهېږي

 زما د زړه دردونه دومره بې دلیله نه دي
 چې ستا د سترګو د بڼو په ژبه نه پوهېږي

 د زړه تڼاکې مې په داسې بڼو وڅیرلې
 چې د تڼاکو د چاودو په ژبه نه پوهېږي

 هغه د خپل ظالم نظر د زوره نه دے خبر
 هغه زما د اسوېلو په ژبه نه پوهېږي

 هغه بې لوظه بې پښتو هغه بې خوږو سړے
 هغه د مینې د پښتو په ژبه نه پوهېږي

 د مصلحت د فلسفو په ژبه نه پوهېږي
د ښائستونو د سپرلي باد ته ګُلونه کرو
 پخپلو زړونو کښې د مینې ارمانونه کرو

 ستا د بڼو سرونه هغسې په زړه ګرځوو
 مونږ د ځیګر په زمکه هغسې زخمونه کرو

 د وخت قاضي دې پرې څه نوې فتوی ولګوي
 مونږ په دې خاوره نوے خیال نوي فکرونه کرو

 زمونږ خودسره منصورخېلو نسل نه ورکېږي
 مونږ د دارونو په سرونو کښې سرونه کرو

 مونږ د نفرت د طوفانونو پرواه نه ده کړې
 مونږه د زمکې په سینه کښ اُلفتونه کرو

 دردونو لږه غوندې ساه واخلۍ چې ساه واخلمه
 زه او ماښام زما په زړه د چا یادونه کرو

 د هغه حسن د سپرلي یاد به ساتو صابره
 په پرهرونو کښې ګُلونه که رنګونه کرو
ستا لېونے یم انتظار مې کوه دربه شمه
 زه لا ژوندے یم انتظار مې کوه دربه شمه

 دلته پردے ښار، د پردو خلقو د نه زړه کلي
 دلته پردے یم انتظار مې کوه دربه شمه

 لا خو زلمے یم انتظار مې کوه دربه شمه

 کلک د وعدې یم انتظار مې کوه دربه شمه

 چې ستا کوڅه یې همېشه د ژوند جنت ګڼلے
 هغه سړے یم انتظار مې کوه دربه شمه

 د تورو غرونو سپينو واورو نه پناه لالیه
 ستا لېونے یم انتظار مې کوه دربه شمه

 د جانان یاده ته ورځه د جانان خوا کښې اوسه
 زه درپسې یم انتظار مې کوه دربه شمه

 ستا د یادونو سيلۍ واخستمه در یې وړمه
 بس درنزدې یم انتظار مې کوه دربه شمه

 د پردي ښار مسافرۍ مې وجود خوري صابره
 په زړه نرے یم انتظار مې کوه دربه شمه
خپل پرهرونه کړو ډيوې خلقو
 راځۍ چې سترګې کړو خوږې خلقو

 ورکې کړۍ خپلې مجبورۍ ورکې کړۍ
 ماتې کړۍ خپلې زولنې خلقو

 ما ستاسو مینه ستاسو پت پاللے
 ما دُنیا پېژني په دې خلقو

 ما د سندرو په تار وپيېلې
 ستاسو د ژوند خوږې قېصې خلقو

 زه د فتوو په دُهند کښې نه پټېدم
 زما خبرې پېژندې خلقو

 څه چې ووم ستاسو د نظره وومه
 څه چې مې وې ستاسو مې وې خلقو

 مه کوۍ ما او جانان مه بېلوۍ
 د یار د کلي د کوڅي خلقو
 چې ایمان مې په ښائست دے چې یقین مې په سپرلي دے
 د چمن غم زما غم دے که په ګُل که په ازغي دے

 د سپرلي تمه ګواه ده چې زما د پرهر فکر
 د ازغو په څوکو اوسي د ګُلونو ارماني دے

 د دُنیا بلا غریب دي د دُنیا بلا مفلس دي
 که د دې ښار محتسب دے که په دې ښار کښې ساقي دے

 څه ناشوني حادثه ده څه نادره المیه ده
 زما ژبه د رندانو، اورېدونکے مې ولي دے

 په ځان هغسې بې وسه، په زور هغسې غاوره
 انسان هغسې نیمګړے انسان هغسې وحشي دے

 که تا نه اورم کڼېږم که ځان نه وينم ورکېږم
 اتفاق مې مجبوري ده اختلاف مې فطرتي دے

 د رڼا په تار تړلے د قامت په صلیب زانګي
 صابر هغسې سرکش دے صابر هغسې باغي دے
څه ګُلونه په شیشو کښې څه ګُلونه د شیشو دي
 ننني ښکلي مخونه، ښائستونه د شیشو دي

 ننني زخمونه څه دي، ننني دردونه څه دي
 تصويرونه په شیشو کښې تصويرونه د شیشو دي

 نه ترې یو خوا څوک پټېږي، نه ترې بل خوا پټېدے شي
 د نیتونو دیوالونه، دېوالونه د شيشو دي


 د جونګړو ډېوې هسې په جونګړو کښې زړې شوې
 د یارانو ارمانونه محلونه د شيشو دي

 خاموشۍ پکښې دېره دي، ویرانۍ پکښې اوسېږي
 که کورونه دي د خاورو، که کورونه د شيشو دي

 د دې ښار غوښې رګونه، د دې ښار خوږمنې سترګې
 سړکونه د تارکولو، دُکانونه د شيشو دي
 که غزل مې ورته ووې که رنا مې ورته ووې
 خپل ایمان مې ورته ووې خپله ساه مې ورته ووې

 تورې سترګې په خندا شوه تورې زلفې په شاه لاړه
 چې بیګاه مې خوا له راغله چې بیګاه مې ورته ووې

 خپل نصیب ته فکر یوړم خپلې برخې ته حیران شوم
 چې اشنا یې راته ووې چې اشنا مې ورته ووې

 اوس یو بل ته کتے نه شو اوس د یو بل نه شرمېږو
 چې رښتیا یې راته ووې چې رښتیا مې ورته ووې

 څه کچه شوه څه خفه شوه مخ یې واړوو غصه شوه
 چې وعده مې کړه وریاده بې وفا مې ورته ووې

 د یوې لارې پښتون یم، په یوه ژبه پوهېږم
 چې تپوس یې رانه وکړو، د زړه غلا مې ورته ووې

 یار مغل وو پښتو نه شو، زه پښتون ووم مغل نه شوم
 د دُنیا یې راته ووې، د دُنیا مې ورته ووې
 د ګُلونو د ښار خلق د رنګونو د ښار خلق
 په خیالونو کښې مې اوسي د خیالونو د ښار خلق

 په رګونو کښې مې ګرزي په زخمونو کښې مې اوسي
 د ارمان د بڼ ګُلونه د یادونو د ښار خلق

 په ما هغه ورځ اختر وي، زما هغه خوشحالي وي
 چې په وېښه چرته وینم د خوبونو د ښار خلق

 زه چې ستا کوڅې له راشم ځه چې ستا دېوال ته کېنم
 په بتکو پورې ولاړ وي د پارونو د ښار خلق

 ليوني تندې اخستي لېوني لېوني ګرزي
 د اوبو د کلي خلق د سیندونو د ښار خلق

 د آسمان په تمه ناستو د آسمان تمه پَرې کړۍ
 د آسمان تمبې وژلي په کلونو د ښار خلق

 دې بې نومه خاموشۍ له نوموړے غم پکار دے
 چې بې خوده ليوني شي د غمونو د ښار خلق

 د دردونو ناخبره د صابره څه خبر دي
 د صابر په زړه پوهږي د دردونو د ښار خلق
د احتساب د سر نشه ماتووم
 په تنده نه یم خو توبه ماتووم

 د انتظار د خاطر صبر تمام
 سر د جانان په دروازه ماتووم

 د مجبورۍ هره سودا خورووم
 د بې وسۍ هره پنجره ماتووم

 تا ته یې یو څه ما ته بل څه ووې
 دا دروغژنه آئينه ماتووم

 د محبت د ارمان هره تنده
 ستا د قامت په تاتره ماتووم

 ورېځې راغلې مېخانې ته ځمه
 خدایه توبه مې ده توبه ماتووم

 ساقي په سترګو کښې نشه راکوه
 پیاله دې سر وخوري پیاله ماتووم
د انتظار جذبه سړه ښه نه ده
 د پرهر څړيکه مې اوده ښه نه ده

 د غرمې تاو ښه نه دے توره به شې
 د غرمې مه راځه غرمه خه نه ده

 واعظه مه کوه ستومانه به شم
 د ځوانۍ ورځې دي توبه ښه نه ده

 د تور اُربل توره ما مه یروه
 زه کله روغ وومه که شپه ښه نه ده

 د محبت لېوني چا وژلي
 د اشنا هسې حوصله ښه نه ده

 شونډې دې رپي چې خبرې کوې
 قاصده ښکاري چې قیصه خه نه ده

 د سبا ستوري ته څه ونه وايي
 بله ډیوه بله لمبه ښه نه ده

 نن به دې زړه ځان سره وړي صابره
 نن یې د سترګو اراده ښه نه ده
د ژوند آهنګ د ژوند نغمه بدلوي
 حالات بدلېږي زمانه بدلوي

 څوک چې په ژوند کښې څه بدلون غواړي
 اول به خپله رویه بدلوي

 هر څوک د خپل ساقي د نومه ستړے
 هر سړے خپله مېخانه بدلوي

 اوس خو تیارې هم ده رڼا د ویرې
 هر شپه نوې لوپټه بدلوي

 وخت چې بدل شي نظریه نوي شي
 ژوند چې بدلېږي فلسفه بدلوي

 په یوه منډه بدلېدونکي دُنیا
 د زندګۍ هره رشته بدلوي

 مونږ لا په زمکه د شپې ځائے نه مومو
 خلق سپوږمۍ ته ځي نشه بدلوي

 جانان یې هغه انتظار یې هغه
 صابر د مېنې طریقه بدلوي
په پرهرونو کښې سپرلي ساتمه
 ژوندي غمونه دې ژوندي ساتمه

 زه پې پرواه یم خو پرواه مه کوه
 د ګودر لاره به سهي ساتمه

 په زړه دننه اسوېلي ساتمه

 هغه کوڅه هغه بتکي ساتمه

 د ارمانون د سپرلي په تمه
 شپلونه څارمه ازغې ساتمه

 د محتسب د نیته ویرېدم
 نور به په سترګو کښې ساقي ساتمه
دلته امکان دے د سپرلي د سحر

 د زړه زخمونه چې ورځي د سحر

ګران دي په ګرانه لاره تله د شعور
