ډېرې وعدې ښې دی لږې لنډې وي
اوس دې هر يو کال راته وعده شوله
تا چې د خاطر خاطر ونه کړلو
ښه شوه ډېره ښه شوه اشنا ښه شوله
بیا به سره ګورو اوس ځوانی ده درومه درومه
څوک کړی دومره غم خو دیوانی ده درومه درومه
اوس ته ايله راغلي چي بس وخت د ځنکدن شو
بيا راته ګراېږي آساني ده درومه درومه
وې يې چې بازار مې د ښائست دے ګرم شوے
هر شے پکښې ختلے دے ګرانی ده درومه درومه
زه او ته خاطره نور په څه سره خواګه یو
مینه خوهم دا پښېمانی ده درومه درومه
شکر چی زه هم ستا ده ښیرې شومه
اوس دې چې ځه ښکليه د کوڅې شومه
ځه به په ځان څه وکړم چې راشې ته
واړه دې په طمعه کښې ذرې شومه
دومره ور نزدې يم که څو لرې یم
لری به شم نور که ور نزدې شومه
نن خو دې خاطره چې سبا نه وے
پوه دې زه د مکر په وعدې شومه
کله چې په فکر کښې د ستا شمه
لاړ شمه له ځانه خو بيا راشمه
ته که لري ضد ځه خو پښتو لرم
اوس در سره څرنګه پخلا شمه
خیال او تصور ته به سلام وکړم
ځه که د خپل زړه په تماشا شمه
تاو مې شی د زړه نه ستا د مخ لمبې
ګډ چی ستا د زلفو په سودا شمه
نورې دې د مينې لارې سمې کړه
خدائے خبر چې کوم پلو خطا شمه
عشقه داسی ځائے لره دې راوستم
اوس به نه د ستا نه د دنیا شمه
نن دې د پرون وه ده د راتلو وعده
نن څنګه په طمع د سبا شمه
ډېر یې غريو نيولى را كتلى دى
لږه به خاطر ته په خندا شمه
دلاری خاوره شوم ده پښو ځنې وتلے يمه
اشنا په تا پسې له تلو ځنې وتلے يمه
د آشنا مخه نور می سوے ژړولے نه شې
زه خود اور او د اوبو ځنې وتلے يمه
په ښه ښیره به دې ځه لږ ډېر نازیدمه اشنا
زړه مې اوس پوه دی د ښېرو ځني وتلے يمه
تا خو به نور را ته خندلی وې چې دا څه وشو
که په ژړا دې د کوڅو ځنې وتلے يمه
دغه داغونه مې دی بس دومره خاطر مې وشو
نور د کرلو د ګلو ځنې وتلے يمه
توبه څنګه تېریږي په هوښيار باندې وختونه
ناپوهو هسې تېر کړل په قرار باندې وختونه
هچرې هم سخیان خو د سبا غمونه نه کا
له شومانو نه تېریږی په خروار باندې وختونه
دا تېر وختونه پرېږده له انکار سره چې تېر شو
ته راشه نور دې سم کړه په اقرار باندې وختونه
نه دے هسی ګرم چی څوک په غم کښې دروند ښکاریږی
په هر چا باندې تېر شی په خپل وار باندې وختونه
ګوره د هغو په زړو کښې نوره سودا نه وی
تیرچی شی د ښکلو په بازار باندې وختونه
لکه باد هسې تيرېږي وخت په روغ او په بیمار
ودرېګي نه ودرېګي په بیمار باندې وختونه
د مخ دې تصور کړم ځکه مخ مې داسې تور دے
خبر نه يې ، کړل تېر مې په انګار باندې وختونه
ما څنګه به لټون کړے د دنیا له غم نه خلاص يم
خاطر ما ښنکلی تېر كړل په خپل یار باندې وختونه
ما په مثال لکه پتنګ وګڼه
ته دې د شمعي غوندې رنګ وګڼه
که می کوثر شولو نصیب
بيا مې شهید دعشق په ننګ وګڼه
زمکه بستر وی لېونو ته مدام
زړګيه ته يې هم پالنګ وګڼه
لکه چې څومره ستا غمونه دی ډېر
پراته مې ټول په زړګي زنګ وګڼه
خاطر کچکول د ګدا واخستلو
اشنا دیدن په ځان قلنګ وګڼه
د ستا د خولې چې فېصله ماته شوه
نوره د زړۀ مې حوصله ماته شوه
شونډې دې هسې رپېدلې نه وې
په زړه دې بله څه راتله ماته شوه
نور مې ساقی څه جرم نہ دے کړے
تشه له لاسه مې پياله ماته شوه
که په عوض کښې يې هر څه غواړې ته
دا می په برخه رسیده ماته شوه
لاس مې بې واکه چې جامونو ته ځي
ماښام نه دے چې روژه ماته شوه
خاطره پرېده د دروغو توبې
نوره زما او ستا ناغه ماته شوه
نور زهر په خوګو كښې د افسون مه راکوه
دا بس دے اشنا نور دا زړه چاودون مه راکوه
ته تل کړه ستمونه خوپه خپله اے اشنا
سزا د رقیبانو په لمسون مه راکوه
چې څو وکړې وعدې نو هــومـره لاشې بې وعدې
دا دې وی در پاتې خو زبون مه راکوه
که ځه چرې پۀ خود وی نو په تا شیدا کېدم
په سترګو ماله دار بیا د جنون مه راکوه
هر څوکه زه خاطره د اشنا اجل ته مخ شوم
د کور مبارکی خوپه ژوندون مه راکوه
ته خو وايه زه به څه کړم زړګيه ليونه
څه به کړم او سه به نه کړم اے زړګيه لېونيه
تا خو وې آبادی پريده ، ما خو پرېښوه دلته څه دی
په صحرا کی ګرځیده کړم اے زړګيه لېونيه
چې اشنا کړی سترګې پورته باڼه ستا په لور رادرومې
زه يې کوم طرف کاږه کړم اے زړګيه لېونيه
زه خاطر درته حیران يم چې په سه باندې راويښ شوې
چې په څه دې اوس وده کړم اے زړګيه لېونيه
ځیګر مي شو کباب په لېمو اور ولګیده
په ما دے یک تنها که په هر لور اولګېده
چې لرې در نزدې شوم د وصال په تمنا
ستا د ګنګروګانو په ما شور ولګېده
ځيګر مې تش ګوګل دے راشه وکړه تماشه
د سوو فریادونو ټنګ ټکور ولګېده
زړګيه اميد لنډ کړه پس له دې نور له وصاله
بېلتون دې ځوانی مرګ شی اوس یې زور ولګېده
سپين مخ دې را ښکاره کړه د نقابه لکه نمر
وربل دې پرې د ورېځو غوندې خور ولګېده
خاطره خداې خبر که بیا دیدن و کړې د ياره
چی بیا د غمازانو داسې زور و لګېده
اشنا وې دا لوګی دی د خاطر د اسويلو
رقيب وې مازیګر شو په تن اور ولګېده
لوئے دریاب د مینې نه شی په واړه زړه
شومه ياره بیا دا ستا غوندې په نه زړه
چي په خله کوی یوه او په زړه بله
د هغه سره به مينه کړې په څۀ زړه
سوی رنګ ته مې لږ وګوره دلبره
ما نيولے ستا د حسن په لمبه زړه
بېګا ستا په انتظار کښې شمع رويه
د هر ستوري سره پورته مې خاته زړه
بيا يې خوب واړه ټوپونه او هېبت شي
د هجران چغې چې وشی په وده زړه
يوغم نه دے چې يې وکړې اے خاطره
ډېر غمونه دی کوه یې په ساړه زړه
داسې بله شپه به په نصیب سره راځی
په عمر کښې راځی خو د رقیب سره راځي

رادرومی خوشحالۍ دې د حبيب سره راځی
څو چې په تندی یې د ازل لیکلی نه وی

خندیږی او ګډيږی سره ګلونه د سبا
